def transform_freq(indic_dict, lst, current_freq):
    dates = logrtn_dict['xlf log return'].index
    
    for i in lst:
        indic_dict[i.split()[0] + ' daily'] = pd.DataFrame(index = dates)
        indic_dict[i.split()[0] + ' daily'][indic_dict[i].columns[0] + ' DAILY'] = pd.Series(None, dtype = 'float')
        
        if current_freq == 'daily':
            for j in indic_dict[i.split()[0] + ' daily'].index:
                val = indic_dict[i][indic_dict[i].columns[0]].loc[(indic_dict[i].index.day == j.day) &
                                                                  (indic_dict[i].index.month == j.month) & 
                                                                  (indic_dict[i].index.year == j.year)][0]

                indic_dict[i.split()[0] + ' daily'][indic_dict[i].columns[0] + ' DAILY'].loc[(indic_dict[i.split()[0] + ' daily'].index.day == j.day) & 
                                                                                             (indic_dict[i.split()[0] + ' daily'].index.month == j.month) & 
                                                                                             (indic_dict[i.split()[0] + ' daily'].index.year == j.year)] = val 

        if current_freq == 'weekly':
            for j in indic_dict[i.split()[0] + ' daily'].index:
                val = indic_dict[i][indic_dict[i].columns[0]].loc[(indic_dict[i].index.month == j.month) & 
                                                                  (indic_dict[i].index.year == j.year)][0]

                indic_dict[i.split()[0] + ' daily'][indic_dict[i].columns[0] + ' DAILY'].loc[(indic_dict[i.split()[0] + ' daily'].index.month == j.month) & 
                                                                                             (indic_dict[i.split()[0] + ' daily'].index.year == j.year)] = val/5
        
        if current_freq == 'monthly':
            for j in indic_dict[i.split()[0] + ' daily'].index:
                val = indic_dict[i][indic_dict[i].columns[0]].loc[(indic_dict[i].index.month == j.month) & 
                                                                  (indic_dict[i].index.year == j.year)][0]

                indic_dict[i.split()[0] + ' daily'][indic_dict[i].columns[0] + ' DAILY'].loc[(indic_dict[i.split()[0] + ' daily'].index.month == j.month) & 
                                                                                             (indic_dict[i.split()[0] + ' daily'].index.year == j.year)] = val/30

        if current_freq == 'quarterly':
            for j in indic_dict[i.split()[0] + ' daily'].index:
                
                if (j.month == 1) or (j.month == 4) or (j.month == 7) or (j.month == 10):
                    val = indic_dict[i][indic_dict[i].columns[0]].loc[(indic_dict[i].index.month == j.month) & 
                                                                      (indic_dict[i].index.year == j.year)][0]
                
                elif (j.month == 2) or (j.month == 5) or (j.month == 8) or (j.month == 11):
                    val = indic_dict[i][indic_dict[i].columns[0]].loc[(indic_dict[i].index.month == j.month - 1) & 
                                                                      (indic_dict[i].index.year == j.year)][0]
                
                else:
                    val = indic_dict[i][indic_dict[i].columns[0]].loc[(indic_dict[i].index.month == j.month - 2) & 
                                                                      (indic_dict[i].index.year == j.year)][0]

                indic_dict[i.split()[0] + ' daily'][indic_dict[i].columns[0] + ' DAILY'].loc[(indic_dict[i.split()[0] + ' daily'].index.month == j.month) & 
                                                                                             (indic_dict[i.split()[0] + ' daily'].index.year == j.year)] = val/90
        
        if current_freq == 'keep same':
            for j in indic_dict[i.split()[0] + ' daily'].index:
                val = indic_dict[i][indic_dict[i].columns[0]].loc[(indic_dict[i].index.month == j.month) & 
                                                                  (indic_dict[i].index.year == j.year)][0]

                indic_dict[i.split()[0] + ' daily'][indic_dict[i].columns[0] + ' DAILY'].loc[(indic_dict[i.split()[0] + ' daily'].index.month == j.month) & 
                                                                                             (indic_dict[i.split()[0] + ' daily'].index.year == j.year)] = val
